def todo_list_manager():
    """A simple command-line To-Do List Manager."""
    tasks = []
    
    while True:
        # Display the menu
        print("\n--- To-Do List Manager ---")
        print("1. View Tasks")
        print("2. Add Task")
        print("3. Mark Task as Complete")
        print("4. Exit")
        
        choice = input("Enter your choice (1-4): ")

        # 1. View Tasks
        if choice == '1':
            if not tasks:
                print("Your to-do list is empty! 🎉")
            else:
                print("\n--- Current Tasks ---")
                for index, task_data in enumerate(tasks):
                    # task_data is a tuple: (task_description, is_complete)
                    status = "✅" if task_data[1] else "❌"
                    print(f"{index + 1}. [{status}] {task_data[0]}")

        # 2. Add Task
        elif choice == '2':
            new_task = input("Enter the new task: ")
            # Store the task as a tuple: (description, is_complete=False)
            tasks.append((new_task, False))
            print(f"Task '{new_task}' added.")

        # 3. Mark Task as Complete
        elif choice == '3':
            if not tasks:
                print("No tasks to complete!")
                continue
            
            try:
                task_num = int(input("Enter the number of the task to complete: "))
                task_index = task_num - 1 # Convert list number to 0-based index
                
                if 0 <= task_index < len(tasks):
                    # Get the current description and update the 'is_complete' status
                    description = tasks[task_index][0]
                    tasks[task_index] = (description, True)
                    print(f"Task '{description}' marked as complete. ✅")
                else:
                    print("Invalid task number.")
            except ValueError:
                print("Invalid input. Please enter a number.")
                
        # 4. Exit
        elif choice == '4':
            print("Goodbye! Have a productive day! 👋")
            break
            
        # Handle invalid input
        else:
            print("Invalid choice. Please enter a number between 1 and 4.")

# Run the program
if __name__ == "__main__":
    todo_list_manager()
